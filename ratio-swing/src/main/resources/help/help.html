Ratio ${pom.version}

Ratio is a calculator that handles rational numbers and matrix algebra.

Syntax

<expression> ::= <compose>
<compose> ::= <add> { "," <add> | "row" <add> | "col" <add> } 
<add> ::= <mul> { "+" <mul> | "-" <mul> }
<mul> ::= <unary> { "*" <unary> | "/" <unary> }
<unary> * ::= <plus> | <negate> | <identity> | <determiner> | <transpose> | <inverse> | <reduce> | <trace> | <term>
<plus> ::= "+" <unary>
<negate> ::= "-" <unary>
<identity> ::= "I" <unary>
<determiner> ::= "det" <unary> 
<transpose> ::= "trans" <unary> 
<trace> ::= "trace" <unary> 
<inverse> ::= "inv" <unary> 
<reduce> ::= "reduce" <unary>
<term> ::= <integer> | <identifier> | "(" <compose> ")"
